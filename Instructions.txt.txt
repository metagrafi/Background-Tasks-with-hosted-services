Start by opening CacheExp solution

At Solution->CacheExp data folder you may find the script for the local db creation.

CREATE DATABASE iNikitz ON (
  NAME='iNikitz', 
  FILENAME='your\path\CacheExp\CacheExp\data\iNikitz.mdf')

I created 2 tables

CREATE TABLE [dbo].[IPProfiles] (
    [IP]        NVARCHAR (50) NOT NULL,
    [City]      NVARCHAR (50) NULL,
    [Country]   NVARCHAR (50) NULL,
    [Continent] NVARCHAR (50) NULL,
    [Latitude]  FLOAT (53)    NULL,
    [Longitude] FLOAT (53)    NULL,
    PRIMARY KEY CLUSTERED ([IP] ASC)
);
CREATE TABLE [dbo].[Works] (
    [Id]               NVARCHAR (50) NOT NULL,
    [TotalActions]     INT           NULL,
    [CompletedActions] INT           NULL,
    [FailedActions]    INT           NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

The connection string is at appsettings.json
"ConnectionStrings": {
    "DefaultConnection": "Data Source=(localdb)\\MSSQLLocalDB;Initial Catalog=iNikitz;Integrated Security=True;Connect Timeout=30;Encrypt=False;TrustServerCertificate=False;ApplicationIntent=ReadWrite;MultiSubnetFailover=False"
  }
